% \captionbadge — inline colored badge matching the photo overlay marker style.
% Renders as a TikZ node with explicit minimum_width / minimum_height in mm
% so every badge has identical outer dimensions regardless of which digit it
% contains. The font size scales with the badge size (font_pt = size_mm × 1.5,
% computed in Go), keeping the same text-to-badge ratio at any size.
%
% Vertical alignment: the geometric center of the badge sits 0.5ex above the
% surrounding text baseline (i.e. roughly at the visual center of the x-height).
% Aligning baselines instead would put the badge optically below the text,
% because the badge is symmetric around its center while text sits above
% baseline. The 0.5ex offset uses the SURROUNDING font's ex-height, so it
% auto-scales with caption_font_size.
%   #1 = badge size (mm) — both width and height
%   #2 = font size (pt) — scaled in Go from #1
%   #3 = background color hex (without #)
%   #4 = text color name (white|black) — chosen via WCAG luminance in Go
%   #5 = marker number
\newcommand{\captionbadge}[5]{%
  {\definecolor{cbbg}{HTML}{#3}%
   \tikz[baseline={([yshift=-0.5ex]badge.center)}]{%
     \node[fill=cbbg,text=#4,inner sep=0,
           minimum width=#1mm,minimum height=#1mm,
           font=\fontsize{#2}{#2}\selectfont\bfseries,
           rectangle](badge){#5};%
   }}%
}
% \captionbadgefb — fallback badge when no chapter color is set
% (matches the overlay fallback of black at 55% opacity). Same fixed-dimension
% TikZ implementation and vertical alignment as \captionbadge.
%   #1 = badge size (mm)
%   #2 = font size (pt)
%   #3 = marker number
\newcommand{\captionbadgefb}[3]{%
  \tikz[baseline={([yshift=-0.5ex]badge.center)}]{%
    \node[fill=black!55,text=white,inner sep=0,
          minimum width=#1mm,minimum height=#1mm,
          font=\fontsize{#2}{#2}\selectfont\bfseries,
          rectangle](badge){#3};%
  }%
}
